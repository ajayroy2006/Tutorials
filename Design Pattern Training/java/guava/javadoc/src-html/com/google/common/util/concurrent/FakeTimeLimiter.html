017package com.google.common.util.concurrent;
018
019import static com.google.common.base.Preconditions.checkNotNull;
020
021import com.google.common.annotations.Beta;
022
023import java.util.concurrent.Callable;
024import java.util.concurrent.TimeUnit;
025
026/**
027 * A TimeLimiter implementation which actually does not attempt to limit time
028 * at all.  This may be desirable to use in some unit tests.  More importantly,
029 * attempting to debug a call which is time-limited would be extremely annoying,
030 * so this gives you a time-limiter you can easily swap in for your real
031 * time-limiter while you're debugging.
032 *
033 * @author Kevin Bourrillion
034 * @since 1.0
035 */
036@Beta
037public final class FakeTimeLimiter implements TimeLimiter {
038  @Override
039  public <T> T newProxy(T target, Class<T> interfaceType, long timeoutDuration,
040      TimeUnit timeoutUnit) {
041    checkNotNull(target);
042    checkNotNull(interfaceType);
043    checkNotNull(timeoutUnit);
044    return target; // ha ha
045  }
046
047  @Override
048  public <T> T callWithTimeout(Callable<T> callable, long timeoutDuration,
049      TimeUnit timeoutUnit, boolean amInterruptible) throws Exception {
050    checkNotNull(timeoutUnit);
051    return callable.call(); // fooled you
052  }
053}
