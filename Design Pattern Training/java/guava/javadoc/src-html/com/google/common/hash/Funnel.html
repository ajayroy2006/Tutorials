015package com.google.common.hash;
016
017import com.google.common.annotations.Beta;
018
019import java.io.Serializable;
020
021/**
022 * An object which can send data from an object of type {@code T} into a {@code PrimitiveSink}.
023 * Implementations for common types can be found in {@link Funnels}.
024 *
025 * <p>Note that serialization of {@linkplain BloomFilter bloom filters} requires the proper
026 * serialization of funnels. When possible, it is recommended that funnels be implemented as a
027 * single-element enum to maintain serialization guarantees. See Effective Java (2nd Edition),
028 * Item 3: "Enforce the singleton property with a private constructor or an enum type". For example:
029 * <pre>   {@code
030 *   public enum PersonFunnel implements Funnel<Person> {
031 *     INSTANCE;
032 *     public void funnel(Person person, PrimitiveSink into) {
033 *       into.putUnencodedChars(person.getFirstName())
034 *           .putUnencodedChars(person.getLastName())
035 *           .putInt(person.getAge());
036 *     }
037 *   }}</pre>
038 *
039 * @author Dimitris Andreou
040 * @since 11.0
041 */
042@Beta
043public interface Funnel<T> extends Serializable {
044
045  /**
046   * Sends a stream of data from the {@code from} object into the sink {@code into}. There
047   * is no requirement that this data be complete enough to fully reconstitute the object
048   * later.
049   *
050   * @since 12.0 (in Guava 11.0, {@code PrimitiveSink} was named {@code Sink})
051   */
052  void funnel(T from, PrimitiveSink into);
053}
